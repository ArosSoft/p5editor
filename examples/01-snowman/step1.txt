/*
 * Функция setup() выполняется один раз в начале программы.
 * Используется для начальных настроек.
 */
function setup() {
  // createCanvas(ширина, высота) - создает область рисования
  // Параметры: ширина и высота в пикселях
  createCanvas(400, 400); // Холст 400x400 пикселей
}

/*
 * Функция draw() выполняется постоянно (60 раз в секунду).
 * Содержит код для рисования каждого кадра.
 */
function draw() {
  // background(кр, зел, син) - заливка фона цветом в модели RGB
  // Значения компонентов: 0-255 (0 - нет цвета, 255 - макс. яркость)
  background(135, 206, 235); // RGB: голубое небо
  
  // fill(яркость) или fill(кр, зел, син) - устанавливает цвет заливки
  // Все последующие фигуры будут залиты этим цветом
  fill(255); // Белый цвет (255, 255, 255)
  
  // noStroke() - отключает рисование контура фигур
  // Противоположность: stroke(цвет) - включает контур
  noStroke();
  
  // Далее следует код рисования снеговика
  // (эллипсы, треугольники, линии и т.д.)
}
