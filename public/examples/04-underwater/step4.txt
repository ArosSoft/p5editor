function draw() {
  // Вода (градиент)
  for (let y = 0; y < height; y++) {     // Идём сверху вниз по пикселям
    let blue = map(y, 0, height, 255, 50);  // Чем ниже, тем темнее синий
    stroke(0, 0, blue);                     // Устанавливаем цвет линии (только синий)
    line(0, y, width, y);                   // Рисуем горизонтальную линию
  }
}
Что это: Рисуем фон.
 Проходим по каждому пикселю сверху вниз
 и делаем цвет от светло-синего (255) сверху
 до тёмно-синего (50) снизу.
