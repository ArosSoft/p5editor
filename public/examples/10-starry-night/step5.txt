//1. Рисование падающих звёзд
//Здесь программа работает со списком shootingStars (падающие звёзды). Перебирает их с конца, чтобы удобно удалять те, что улетели:

//1.1. У каждой падающей звезды есть координаты (x, y), скорость (vx, vy) и «жизнь» (life) — сколько ещё осталось быть видимой.

//1.2. На каждом кадре координаты увеличиваются на скорость — звезда движется.

//1.3. «Жизнь» уменьшается на 2, и когда она станет 0 — звезда исчезает (удаляется из списка).

//1.4. Рисуется белая линия (хвост) от текущего положения до предыдущего (это создаёт эффект движения).

for (let i = shootingStars.length - 1; i >= 0; i--) {
    let s = shootingStars[i];
    s.x += s.vx;
    s.y += s.vy;
    s.life -= 2;
    
    if (s.life <= 0) {
      shootingStars.splice(i, 1);
    }
    
    stroke(255, 255, 255, s.life);
    strokeWeight(2);
    line(s.x, s.y, s.x - s.vx * 2, s.y - s.vy * 2);
  }
