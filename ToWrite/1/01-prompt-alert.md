# Ввод-вывод данных: `prompt()`, `alert()`, `console.log()`

## 📌 Что это такое?

Эти команды помогают программе **общаться с пользователем**:

| Команда | Зачем нужна |
|---------|-------------|
| `prompt("Вопрос")` | Спросить у пользователя что-то и получить ответ |
| `alert("Текст")` | Показать важное сообщение в окошке |
| `console.log("Текст")` | Написать сообщение в консоль (для разработчика) |

---

## Пример 1: Простое приветствие

**Файл:** `sketch_01.js`

```javascript
function setup() {
  createCanvas(400, 400);
  
  // Спрашиваем имя
  let name = prompt("Как тебя зовут?");
  
  // Показываем приветствие
  alert("Привет, " + name + "!");
  
  // Пишем в консоль для проверки
  console.log("Пользователь:", name);
}

function draw() {
  background(220);
  
  // Рисуем приветствие на холсте
  textAlign(CENTER, CENTER);
  textSize(24);
  fill(0);
  text("Привет, " + name + "!", width/2, height/2);
}
```

### 🔍 Как это работает:

1. `prompt()` открывает окошко с вопросом
2. Ответ сохраняется в переменную `name`
3. `alert()` показывает сообщение с именем
4. `console.log()` пишет имя в консоль (нажми F12 в браузере)

## Пример 2: Калькулятор

**Файл:** `sketch_02.js`

```javascript
let result;

function setup() {
  createCanvas(400, 400);
  
  // Спрашиваем два числа
  let num1 = prompt("Введи первое число:");
  let num2 = prompt("Введи второе число:");
  
  // Превращаем строки в числа
  num1 = Number(num1);
  num2 = Number(num2);
  
  // Считаем сумму
  result = num1 + num2;
  
  // Показываем результат
  alert("Сумма: " + result);
  
  console.log(num1, "+", num2, "=", result);
}

function draw() {
  background(220);
  
  textAlign(CENTER, CENTER);
  textSize(20);
  fill(0);
  
  text(num1 + " + " + num2 + " = " + result, width/2, height/2);
}
```

### ⚠️ Важно:

`prompt()` всегда возвращает **строку**! Нужно превращать в число через `Number()`.

## Пример 3: Проверка ответа (викторина)

**Файл:** `sketch_03.js`

```javascript
let isCorrect;

function setup() {
  createCanvas(400, 400);

  // Задаём вопрос
  let answer = prompt("Сколько будет 5 × 5?");

  // Проверяем ответ
  isCorrect = (answer === "25");

  // Показываем результат
  if (isCorrect) {
    alert("Правильно! 🎉");
    console.log("✅ Верный ответ");
  } else {
    alert("Неправильно! Правильный ответ: 25");
    console.log("❌ Ошибка:", answer);
  }
}

function draw() {
  background(isCorrect ? 100 : 200);

  textAlign(CENTER, CENTER);
  textSize(24);
  fill(255);

  if (isCorrect) {
    text("✅ Молодец!", width/2, height/2);
  } else {
    text("❌ Попробуй ещё!", width/2, height/2);
  }
}
```

## Пример 4: Ввод нескольких чисел (массив)

**Файл:** `sketch_04.js`

```javascript
let numbers = [];
let sum = 0;
let avg = 0;

function setup() {
  createCanvas(400, 400);

  // Спрашиваем числа (можно через пробел или запятую)
  let input = prompt("Введи числа через пробел (например: 5 10 15 20):");

  // Разбиваем строку на части по пробелам или запятым
  // .split(/[\s,]+/) — разбить по пробелам (\s) или запятым (,)
  // .filter(n => n !== "") — убрать пустые значения
  let parts = input.split(/[\s,]+/).filter(n => n !== "");

  // Превращаем каждое значение в число
  numbers = parts.map(n => Number(n));

  // Считаем сумму
  sum = 0;
  for (let i = 0; i < numbers.length; i++) {
    sum = sum + numbers[i];
  }

  // Считаем среднее
  avg = sum / numbers.length;

  // Показываем результат
  alert("Чисел: " + numbers.length + "\nСумма: " + sum + "\nСреднее: " + avg);

  console.log("Введённые числа:", numbers);
  console.log("Сумма:", sum);
  console.log("Среднее:", avg);
}

function draw() {
  background(220);

  textAlign(CENTER, CENTER);
  textSize(18);
  fill(0);

  // Рисуем введённые числа
  text("Числа: " + numbers.join(", "), width/2, height/2 - 30);
  text("Сумма: " + sum, width/2, height/2 + 10);
  text("Среднее: " + avg.toFixed(2), width/2, height/2 + 50);
}
```

### 🔍 Как это работает:

1. `prompt()` получает строку вроде `"5 10 15 20"` или `"5,10,15,20"`
2. `.split(/[\s,]+/)` разбивает строку на части по **пробелам** или **запятым**
3. `.filter(n => n !== "")` убирает пустые значения
4. `.map(n => Number(n))` превращает каждую строку в число
5. Получается массив чисел: `[5, 10, 15, 20]`

### 💡 Другие способы:

**Через запятую:**
```javascript
let input = prompt("Введи числа через запятую:");
let numbers = input.split(",").map(n => Number(n.trim()));
```

**В отдельные переменные:**
```javascript
let input = prompt("Введи 3 числа через пробел:");
let parts = input.split(" ");
let a = Number(parts[0]);
let b = Number(parts[1]);
let c = Number(parts[2]);
```

## 🎯 Запомни:

| Команда | Что делает | Когда использовать |
|---------|------------|-------------------|
| `prompt()` | Спрашивает и возвращает ответ | Когда нужно получить данные от пользователя |
| `alert()` | Показывает сообщение | Для важных уведомлений |
| `console.log()` | Пишет в консоль | Для отладки и проверки значений |
